```
GET /api/sites/:site/user-traits/values
```

Returns the distinct values for a given trait key, with user counts.

### Path Parameters

<TypeTable
  type={{
    site: {
      description: 'Site ID or site identifier',
      type: 'string | number',
      required: true,
    },
  }}
/>

### Query Parameters

<TypeTable
  type={{
    key: {
      description: 'The trait key to list values for',
      type: 'string',
      required: true,
    },
    limit: {
      description: 'Maximum number of values to return',
      type: 'number',
      default: '1000',
    },
    offset: {
      description: 'Number of values to skip',
      type: 'number',
      default: '0',
    },
  }}
/>

### Response

<TypeTable
  type={{
    values: {
      description: 'Array of distinct trait values',
      type: 'TraitValue[]',
    },
    total: {
      description: 'Total number of distinct values',
      type: 'number',
    },
    hasMore: {
      description: 'Whether more values are available beyond this page',
      type: 'boolean',
    },
  }}
/>

#### TraitValue Object

<TypeTable
  type={{
    value: {
      description: 'The trait value',
      type: 'string',
    },
    userCount: {
      description: 'Number of users with this value',
      type: 'number',
    },
  }}
/>
