```
GET /api/sites/:site/user-traits/users
```

Returns users matching a specific trait key and value, with location and session metadata.

### Path Parameters

<TypeTable
  type={{
    site: {
      description: 'Site ID or site identifier',
      type: 'string | number',
      required: true,
    },
  }}
/>

### Query Parameters

<TypeTable
  type={{
    key: {
      description: 'The trait key to match on',
      type: 'string',
      required: true,
    },
    value: {
      description: 'The trait value to match on',
      type: 'string',
      required: true,
    },
    limit: {
      description: 'Maximum number of users to return',
      type: 'number',
      default: '50',
    },
    offset: {
      description: 'Number of users to skip',
      type: 'number',
      default: '0',
    },
  }}
/>

### Response

<TypeTable
  type={{
    users: {
      description: 'Array of matching users',
      type: 'TraitUser[]',
    },
    total: {
      description: 'Total number of matching users',
      type: 'number',
    },
    hasMore: {
      description: 'Whether more users are available beyond this page',
      type: 'boolean',
    },
  }}
/>
